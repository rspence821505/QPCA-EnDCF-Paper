\section*{Issues found (Sections 3--4 of \texttt{uq\_paper\_draft.tex})}

\begin{enumerate}[leftmargin=*,itemsep=0.75em]
\item \severity{M}\ \textbf{Section 3.1, Definition~\ref{def:data_space} (Observation space and QoI map).}\\
\textbf{Issue:} The “tangent-linear approximation” is written as \(Q(\mathbf{x}_0)\approx \mathbf{H}^{(L)}\mathbf{x}_0\) while \(\mathbf{M}_\ell\) is defined as a Jacobian evaluated at a reference state \(\bar{\mathbf{x}}_0\). A first-order Taylor/tangent-linear approximation about \(\bar{\mathbf{x}}_0\) is affine in \(\mathbf{x}_0\) (involves an intercept and the increment \(\mathbf{x}_0-\bar{\mathbf{x}}_0\)), so the formula as stated misuses the Jacobian as a map on the full state.\\
\textbf{Potential fix:} Replace the linearization by an explicitly affine TLM statement, e.g.\ \(Q(\mathbf{x}_0)\approx Q(\bar{\mathbf{x}}_0)+\mathbf{H}^{(L)}(\mathbf{x}_0-\bar{\mathbf{x}}_0)\), and define \(\mathbf{H}^{(L)}\) as the Jacobian of \(Q\) (or of the stacked observation map) evaluated at \(\bar{\mathbf{x}}_0\). If the intent is to keep a purely linear map, explicitly re-center the state variable (work with \(\delta\mathbf{x}_0:=\mathbf{x}_0-\bar{\mathbf{x}}_0\)) so that the tangent-linear relation is stated for increments rather than absolute states.\\
\textbf{Category:} Logic / Algebra.

\item \severity{M}\ \textbf{Section 4.5, Corollary~\ref{cor:optimal_kappa_new}, Eq.~\eqref{eq:kappa_proxy_objective} (proxy objective).}\\
\textbf{Issue:} The “projector-estimation” term in \eqref{eq:kappa_proxy_objective} uses \(\bigl(\mathbb{E}\|\mathbf{e}^{(1),s}-\boldsymbol{\mu}_E^{\,s}\|^4\bigr)^{1/2}\). This is not consistent with the bound previously available for the projector factor in Theorem~\ref{thm:bias_variance}, Eq.~\eqref{eq:var_projector_term_bound_main}, which involves \(\bigl(\mathbb{E}\|\bar{\mathbf{e}}^{\,s}-\boldsymbol{\mu}_E^{\,s}\|^4\bigr)^{1/2}\) (sample-mean fourth moment, carrying an extra \(N^{-1}\) scaling). The proof’s claim that the proxy is obtained by “collecting the upper bounds in Theorem~\ref{thm:bias_variance}” therefore does not match the stated objective term, and the \(N\)-scaling is altered.\\
\textbf{Potential fix:} Either (i) change the term in \eqref{eq:kappa_proxy_objective} to match Theorem~\ref{thm:bias_variance} by using \(\bigl(\mathbb{E}\|\bar{\mathbf{e}}^{\,s}-\boldsymbol{\mu}_E^{\,s}\|^4\bigr)^{1/2}\), with its \(N^{-1}\) scaling controlled by Lemma~\ref{lem:sample_mean_fourth_moment}, or (ii) if a per-sample fourth moment is preferred for presentation, insert the missing \(N\)-dependence explicitly (e.g.\ via \(\bigl(\mathbb{E}\|\bar{\mathbf{e}}^{\,s}-\boldsymbol{\mu}_E^{\,s}\|^4\bigr)^{1/2}\lesssim N^{-1}\bigl(\mathbb{E}\|\mathbf{e}^{(1),s}-\boldsymbol{\mu}_E^{\,s}\|^4+(\mathbb{E}\|\mathbf{e}^{(1),s}-\boldsymbol{\mu}_E^{\,s}\|^2)^2\bigr)^{1/2}\)) so the proxy term and its scaling align with the preceding bounds.\\
\textbf{Category:} Logic.

\item \severity{L}\ \textbf{Section 4.5, Corollary~\ref{cor:stochastic_comparison_new}, covariance identity / inequality \eqref{eq:stoch_var_bound_new}.}\\
\textbf{Potential Issue:} The covariance additivity and the lower bound in \eqref{eq:stoch_var_bound_new} require that the perturbations \(\{\boldsymbol{\epsilon}_w^{(j)}\}\) are independent of the forecast ensemble (hence independent of \(\bar{\mathbf{x}}^{a}_{\mathrm{det}}\)). This independence is invoked in the proof (“follows from independence”) but is not explicitly stated in the corollary hypotheses.\\
\textbf{Potential fix:} Add an explicit hypothesis that \(\{\boldsymbol{\epsilon}_w^{(j)}\}_{j=1}^N\) are i.i.d.\ and independent of the forecast ensemble (equivalently, independent of \(\bar{\mathbf{x}}^{a}_{\mathrm{det}}\)). If the intent is to allow random (ensemble-dependent) \(\mathbf{K}^{(w)}\), restate the covariance identity conditionally on the forecast ensemble (or on \(\mathbf{K}^{(w)}\)) and then take expectations to obtain the unconditional injected-variance term.\\
\textbf{Category:} Hypothesis / Probabilistic validity.
\end{enumerate}
